<div>

</div>

<Paragraph4>
  Title section component is used to add styled lines into the typography components.
</Paragraph4>

<Paragraph4>
  To use it is necessary to add the prop <i><b>isHighLight</b></i>.
</Paragraph4>

<div>
  <H3 font="secondary">Props</H3>
  <div style={{ marginTop: '1rem' }}>
    <ul> 
      <li><i><b>isHighLight</b></i>: 'true' | 'false' </li>
      <li>
        <i><b>highLightSize</b></i>: 'small' | 'medium' | 'large'
      </li>
      <li>
       <i><b>highLightAlign</b></i>: 'bottom' | 'top'
      </li>
      <li>
      <i><b>align</b></i>: 'right' | 'left' | 'center'
      </li>
      <li><i><b>highLightKind</b></i>: 'primary' | 'secondary' | 'tertiary  </li>
    </ul>
  </div>
</div>
